常见排序算法

补充：算法的稳定性定义为对于待排序列中相同项的原来次序不能被算法改变则称该算法稳定。

对于稳定性：

  • 冒泡排序：冒泡是将小的元素或大的元素向右交换，是通过相邻元素交换完成的，若两个元素相等是不会交换的，所以相同元素的前后顺序不变，所以稳定
  • 选择排序：给每个位置选择当前元素最小的，用最小的元素和当前位置的元素交换。比如 5 8 5 2 9,索引0处找到是最小元素2，然后和索引0的5和2交换，这时候发现5和5的相对顺序已经改变
  • 插入排序：插入排序是在一个已经有序的小序列的基础上，一次插入一个元素。很明显，在有相同元素时，会放到相同元素的后面，不会改变相对位置
  • 快速排序：在最后枢纽元素和左右指针最后所在的元素交换时，可能会致相同元素的相对顺序改变，如5 3 3 4 3 8 9 10 11，5和3交换，3的相对顺序改变
  • 归并：递归出口是一个元素，主要在于合并过程，合并中，相同元素在左边是不会被放到右边的，所以稳定

对于快排最坏时间复杂度，如果待排序的序列为正序或者逆序，每次划分只得到一个比上一次划分少一个记录的子序列，另一个为空，这样复杂度会变为$O(n^2)$

三种简单排序 选择、冒泡、插入

/**
 * Created with IntelliJ IDEA.
 * Description:选择排序
 * 思想：选择出数组中的最小元素，将它与数组的第一个元素交换位置。
 *       再从剩下的元素中选择出最小的元素，将它与数组的第二个元素交换位置。不断进行这样的操作，直到将整个数组排序。
 *
 */
public class Selection {
    public static void sort(int[] arr){
        int n = arr.length;
        for (int i = 0; i < n-1; i++) {
            // 存放每一趟内循环的最小元素下标
            int min = i;
            for (int j = i+1; j < n; j++) {
                if (arr[j] < arr[min]){
                    min = j;
                }
            }
            // 进行交换 如果min发生变化，则进行交换
            if (min != i){
                swap(arr, i, min);
            }
        }
    }
}

/**
 * Created with IntelliJ IDEA.
 * Description:冒泡排序
 * 思想：从左到右不断交换相邻逆序的元素，在一轮的循环之后，可以让未排序的最大元素上浮到右侧。
 *在一轮循环中，如果没有发生交换，就说明数组已经是有序的，此时可以直接退出。
 *
 *
 */
public class Bubble {
    public static void sort(int[] arr){
        int n = arr.length;
        for (int i = 0; i < n-1; i++) {
            // 标记变量 true表示本次内循环没有进行交换，也就是待排序列已经有序
            boolean flag = true;
            for (int j = 0; j < n-1-i; j++) {
                if (arr[j] > arr[j+1]){
                    swap(arr, j, j+1);
                    flag = false;
                }
            }
            // 本次循环没有交换，排序已完成，退出外循环
            //注意flag应该在外层循环中
            if  (flag){
                break;
            }
        }
    }
}

/**
 * Created with IntelliJ IDEA.
 * Description:插入排序
 * 思想：每次都将当前元素插入到左侧已经排序的数组中，使得插入之后左侧数组依然有序。
 *
 */
public class Insertion {
    public static void sort(int[] arr){
        int n = arr.length;
        for (int i = 1; i < n; i++) {
            // i表示当前需要插入的元素
            int j = i;
            // while循环将arr[i]调整至合适位置
            while (j > 0){
                if (arr[j]<arr[j-1]){
                    swap(arr, j, j-1);
                }
                j--;
            }
        }
    }
    //另外一种写法
    public static void insertSort(int[] nums){
        for (int i = 1; i < nums.length; i++) {
            //认为0~i有序
            int  temp=nums[i];
            int j;
            //采用不断将元素向右移动的方式找到合适的位置
            for ( j = i-1; j >=0; j--) {
                if(nums[j]>temp){
                    nums[j+1]=nums[j];
                }else{
                    //一定要break，否则j就不在所需要的位置上了
                    break;
                }
            }
            nums[j+1]=temp;
        }
    }
}

希尔排序

/**
 * Created with IntelliJ IDEA.
 * Description:希尔排序
 * 希尔排序也是一种插入排序，它解决了插入排序每次只能交换相邻元素的局限性
 * 思想：希尔排序使用插入排序对间隔 h 的序列进行排序。通过不断减小 h，最后令 h=1，就可以使得整个数组是有序的。
 *
 *
 */
public class Shell {
    public static void sort(int[] arr){
        int n = arr.length;
        // 增量gap，并逐步缩小增量
        for (int gap = n/2; gap >0 ; gap/=2) {
            // 从第gap个元素，逐个对其所在组直接进行插入排序
            for (int i = gap; i < n; i++) {
                int j = i;
                while (j-gap >= 0 && arr[j] < arr[j-gap]){
                    swap(arr, j, j-gap);
                    j -= gap;
                }
            }
        }
    }
}

归并排序

/**
 * Created with IntelliJ IDEA.
 * Description:归并排序
 * 思想：归并排序的思想是将数组分成两部分，分别进行排序，然后归并起来。
 *
 */
public class Merge {
    /**
     * 归并函数 用递归的方式把数据细分，最后合并
     */
    public static void mergeSort(int[] arr, int left, int right, int[] temp){
        if (left < right){
            int mid = (left + right)/2;
            mergeSort(arr, left, mid, temp);            //左边归并排序，使得左子序列有序
            mergeSort(arr, mid+1, right, temp);    //右边归并排序，使得右子序列有序
            merge(arr, left, mid, right, temp);         //将两个有序子数组合并操作
        }
    }

    /**
     * 合并函数
     */
    public static void merge(int[] arr, int left, int mid, int right, int[] temp) {
        int i = left;
        int j = mid + 1;
        int t = 0;
        while (i <= mid && j <= right) {
            if (arr[i] <= arr[j]) {
                temp[t++] = arr[i++];
            } else {
                temp[t++] = arr[j++];
            }
        }
        // 将数组arr[left,mid]剩余的部分填充进temp
        while (i <= mid) {
            temp[t++] = arr[i++];
        }
        // 将数组arr[mid+1,right]剩余的部分填充进temp
        while (j <= right) {
            temp[t++] = arr[j++];
        }
        // 将归并后的数值复制给数组arr[left,right]，注意从arr[left]开始赋值
        t = 0;
        while (left <= right) {
            arr[left++] = temp[t++];
        }
        // 复制的时候这样写为什么错
        // for(i=0;i<t;i++){
        //     arr[left++]=temp[i++]; //多写了个i++;
        // }
    }
}

快速排序

快速排序通过一个切分元素将数组分为两个子数组，左子数组小于等于切分元素，右子数组大于等于切分元素，将这两个子数组排序也就将整个数组排序了。

public class QuickSort implements Sort{
    public static void quickSort(int[] arr, int left, int right){
        if (left >=right){
            return;
        }
        int index = partition(arr, left, right);
        quickSort(arr, left,index-1);
        quickSort(arr, index+1, right);
    }

    /**
     * 左右指针法
     * 选最左元素为主元（也可以选最右元素为主元）
     * 步骤：
     * 1. 选取一个关键字(key)作为枢轴，一般取整组记录的第一个数/最后一个，这里采用选取序列最后一个数为枢轴。
     * 2. 设置两个变量left = 0;right = N - 1;
     * 3. 从left一直向后走，直到找到一个大于key的值，right从后至前，直至找到一个小于key的值，然后交换这两个数。
     * 4. 重复第三步，一直往后找，直到left和right相遇，这时将key放置left的位置即可
     *
     * @param a
     * @param left
     * @param right
     * @return
     */
    private static int partition(int[] a, int left, int right){
        // 记录最左元素为主元
        int key = a[left], t = 0;
        int i = left;
        int j = right;
        while(i < j) {
            // 为排除一边倒的情况，若以左边第一个元素为主元，应先扫描j，若以右边第一个元素为主元，先扫描i
            // 先从右向左扫描（必须）
            while(a[j] >= key && i < j) {
                j--;
            }
            // 再从左往右扫描
            while(a[i] <= key && i < j) {
                i++;
            }
            //交换两个数在数组中的位置
            if(i < j)
            {
                t = a[i];
                a[i] = a[j];
                a[j] = t;
            }
        }
        //最终将基准数归位
        a[left] = a[i];
        a[i] = key;
        return i;
    }
}

/**
 * 三数取中法
 * @param a
 * @param left
 * @param right
 * @return
 */
private static int partition2(int[] a, int left, int right){
    dealPivot(a, left, right);
    int pivot = a[right-1], t = 0;
    int i = left;
    int j = right-1;
    while (i < j){
        while (a[i] <= pivot && i < j){
            i++;
        }
        while (a[j] >= pivot && i < j){
            j--;
        }
        if (i < j){
            t = a[i];
            a[i] = a[j];
            a[j] = t;
        }
    }
    a[right-1] = a[i];
    a[i] = pivot;
    return i;
}
/**
 * 选取主元
 * 三数先排序再取中间
 * @param a
 * @param left
 * @param right
 */
private static void dealPivot(int[] a, int left, int right){
    int mid = (left + right)/2;
    if (a[left] > a[mid]){
        swap(a, left, mid);
    }
    if (a[left] > a[right]){
        swap(a, left, right);
    }
    if (a[mid] > a[right]){
        swap(a, mid, right);
    }
    // 取中间数为主元，并放在末尾（倒数第二个）
    swap(a, right-1, mid);
}
